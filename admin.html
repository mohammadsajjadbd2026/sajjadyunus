🚀 Supabase সেটআপ গাইড

supabase.com এ গিয়ে ফ্রি অ্যাকাউন্ট খুলুন এবং একটি নতুন প্রজেক্ট তৈরি করুন।

প্রজেক্টের SQL Editor এ গিয়ে নিচের SQL কোডটি কপি করে চালান — এটি আপনার দুটি টেবিল তৈরি করবে:

-- Posts টেবিল CREATE TABLE posts ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, title TEXT NOT NULL, excerpt TEXT, body TEXT, category TEXT DEFAULT 'other', cover_image TEXT, featured BOOLEAN DEFAULT false, published BOOLEAN DEFAULT false, published_at TIMESTAMPTZ, created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); -- Site settings টেবিল CREATE TABLE site_settings ( key TEXT PRIMARY KEY, value TEXT, updated_at TIMESTAMPTZ DEFAULT now() ); -- Public read access (পাঠকরা পড়তে পারবে) ALTER TABLE posts ENABLE ROW LEVEL SECURITY; ALTER TABLE site_settings ENABLE ROW LEVEL SECURITY; CREATE POLICY "public read posts" ON posts FOR SELECT USING (published = true); CREATE POLICY "anon all posts" ON posts FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "public read settings" ON site_settings FOR SELECT USING (true); CREATE POLICY "anon all settings" ON site_settings FOR ALL USING (true) WITH CHECK (true); -- Image storage bucket INSERT INTO storage.buckets (id, name, public) VALUES ('post-images', 'post-images', true); CREATE POLICY "public images" ON storage.objects FOR SELECT USING (bucket_id = 'post-images'); CREATE POLICY "upload images" ON storage.objects FOR INSERT WITH CHECK (bucket_id = 'post-images');

প্রজেক্টের Settings → API থেকে আপনার Project URL এবং anon public key কপি করুন।
